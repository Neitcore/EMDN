Darba programma:

function y = lab3(t)

if nargin == 0
    t = 0:0.01:5;
end
%%Datu filtracija
%sin
t_sin_f = (t>=4.5) & (t<=5);
t_sin = t(t_sin_f);
%saw
t_saw_f = (t>=0)&(t<0.5);
t_saw = t(t_saw_f);
%const
t_const_f = (t>=3.5)&(t<4.5);
t_const = t(t_const_f);
%zero
t_zero_f = (t>=1.5)&(t<3.5);
t_zero=t(t_zero_f);
%noise
t_noise_f = (t>=0.5)&(t<=1.5);
t_noise=t(t_noise_f);

A0=0; A= 1.5;
T = (5-4.5)/2;f=1/T;
delay = 1.5;
y_sin=A0+A*sin(2*pi*f*(t_sin-delay));
%plot(t_sin,y_sin)

%axis([0 5 -1.5 1.5])
